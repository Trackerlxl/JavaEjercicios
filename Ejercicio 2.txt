package Ejerciciosjava;
import java.util.Scanner;

public class Ejercicio2 {

	public static void main(String[] args) {
		
		
        int cantidadLlantas;
        double precioLlantas;
        double totalCompra;

        
        Scanner scanner = new Scanner(System.in);

        
        System.out.print("Ingrese la cantidad de llantas que desea comprar: ");
        cantidadLlantas = scanner.nextInt();

        
        if (cantidadLlantas < 5) {
            precioLlantas = 100.0;
        } else if (cantidadLlantas >= 5 && cantidadLlantas <= 10) {
            precioLlantas = 75.0;
        } else {
            precioLlantas = 50.0;
        }

        
        totalCompra = cantidadLlantas * precioLlantas;

       
        System.out.println("Precio por cada llanta: U$" + precioLlantas);
        System.out.println("Total a pagar por la compra de " + cantidadLlantas + " llantas: U$" + totalCompra);

        
        scanner.close();
    


	}

}
